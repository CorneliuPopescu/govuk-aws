# PROVISIONING
#

rm -rf /etc/puppet
mkdir -p /etc/puppet
if [ -d /var/lib/puppet/ssl ] ; then
  rm -rf /var/lib/puppet/ssl
fi
cat <<EOF >/etc/puppet/puppet.conf
  [main]
  pluginsync = true

  [agent]
  report = false
  configtimeout = 600
EOF

puppet agent --pluginsync --configtimeout 600 --test --waitforcert 60
